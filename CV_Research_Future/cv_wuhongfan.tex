% ===================== Publications & Working Papers =====================
\section*{Publications \& Working Papers}
\begin{itemize}
  \item \textbf{Spatial Matching with Heterogeneous Demand: Coordinated Key Matching Indices for Ride-Hailing} (submitted), \emph{Operations Research}.\\
  \emph{(with Sirui Wang, Jiheng Zhang)}. Developed a two-class spatial matching model with endogenous pickup times and two-stage customer loss (pre-match abandonment, post-match cancellation); derived \emph{Key Matching Indices (KMI)} for each class that decompose into (i) abandonment reduction, (ii) opportunity cost of consuming idle drivers, and (iii) cross-market externality term; characterized optimal thresholds by the coupled unity condition $(\zeta_1,\zeta_2)=(1,1)$ and designed a Jacobian-based coordinated Newton controller with global convergence in numerical experiments.

  \item \textbf{Airline Cargo Transport Recovery} (working paper).\\
  Stochastic modeling of bellyhold cargo recovery under flight disruptions: formulate backlog clearing with random residual capacities and direct-vs-transfer routing options; propose time-threshold segmentation policies to trade off waiting cost and transfer delay; (manuscript in progress).

  \item \textbf{PriceMoE: Shadow-Price-Based Dynamic Routing for MoE Serving} (working paper). \\
  \emph{Core idea:} formulate routing as network utility maximization with capacity constraints; interpret queue length / dual variables as shadow prices; derive a top-$k$ routing rule that trades off semantic score and congestion price.

  \item \textbf{Fluid-Affinity: Locality-Aware Token Routing for Distributed MoE Inference via Online Stochastic Control} (working paper). \\
  \emph{Core idea:} model cross-device routing as online constrained optimization with transition (communication) costs; derive a Lyapunov drift-plus-penalty control with $[O(1/V), O(V)]$ optimality-delay tradeoff; integrates ``GPU affinity'' reward with backpressure for stability.

  \item \textbf{Topology-aware MoE Routing via Two-stage Re-rank} (paper-ready draft / working paper). \\
  \emph{Core idea:} keep semantic candidate set (top-$M$) unchanged; re-rank within candidates with topology-aware communication cost and load guard; targets multi-node prefill-heavy regimes to reduce cross-node bytes and tail latency; connects to OR-style online control.
\end{itemize}
